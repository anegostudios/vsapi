Examples

"creativeinventoryStacksByType": {
	"*-fired": [
		{
			"tabs": [ "general", "decorative" ],
			"stacks": [
				{
					"type": "block",
					"code": "bowl-fired",
					"attributes": {
						"ucontents": [
							{
								"type": "item",
								"code": "waterportion",
								"makefull": true
							}
						]
					}
				},
				{
					"type": "block",
					"code": "bowl-fired",
					"attributes": {
						"ucontents": [
							{
								"type": "item",
								"code": "honeyportion",
								"makefull": true
							}
						]
					}
				},
				{
					"type": "block",
					"code": "bowl-fired"
				},
				{
					"type": "block",
					"code": "bowl-raw"
				}
			]
		}
	]
},
